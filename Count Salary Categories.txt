with newTable as (select account_id , case when income < 20000 then 'Low Salary' when income between 20000 and 50000 then 'Average Salary' else 'High Salary' end as category 
from Accounts)
, fullCategoriesTable as (
        SELECT 'Low Salary' AS category
    UNION ALL
    SELECT 'Average Salary' 
    UNION ALL
    SELECT 'High Salary'
)

select f.category, count(n.account_id) as accounts_count
from fullCategoriesTable f left join newTable n
on f.category = n.category
group by  f.category
