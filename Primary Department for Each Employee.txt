WITH allWhichY AS (
    SELECT *
    FROM Employee 
    WHERE primary_flag = 'Y'
),
withOneFlag AS (
    SELECT employee_id
    FROM Employee
    GROUP BY employee_id
    HAVING COUNT(*) = 1
)
SELECT employee_id,department_id 
FROM allWhichY
UNION
SELECT e.employee_id ,e.department_id 
FROM Employee e
JOIN withOneFlag w
    ON e.employee_id = w.employee_id;
