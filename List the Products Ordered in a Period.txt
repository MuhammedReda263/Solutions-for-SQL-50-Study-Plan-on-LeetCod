with cte1 as (
    select product_id , sum (unit) as units
    from Orders 
    where order_date between '2020-02-01' and '2020-02-29' 
    group by product_id   
    having sum(unit)>=100)

    select p.product_name , e1.units as unit
    from Products p, cte1 e1
    where p.product_id  = e1.product_id
